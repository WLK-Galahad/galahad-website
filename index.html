import tkinter as tk import itertools # Farver til blink colors = ['red', 'yellow', 'green', 'blue', 'magenta', 'cyan'] # Generator til at cykle igennem farverne color_cycle = itertools.cycle(colors) # Funktion til at opdatere farven def blink(): label.config(fg=next(color_cycle)) root.after(300, blink) # Opdater farven hver 300 ms # Opsætning af vindue root = tk.Tk() root.title("Blinkende tekst") root.configure(bg='black') # Etiket med teksten label = tk.Label(root, text="Hej, du er FUCKING grim", font=("Helvetica", 32, "bold"), bg='black') label.pack(expand=True) # Start blink blink() # Kør vinduet root.mainloop()
